SELECT WORKER_ID,FIRST_NAME,SALARY FROM
(
SELECT WORKER_ID,FIRST_NAME,SALARY,COUNT(*) OVER
(PARTITION by SALARY) AS SALARYCNT
FROM [dbo].[WORKER_DETAILS]
)S1
WHERE SALARYCNT>1
ORDER by SALARY DESC
